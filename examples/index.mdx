## Overview

This section contains practical examples showing how to use the library in real-world scenarios.

## Basic Example

A simple example demonstrating core functionality:

```javascript
import { createClient } from 'your-library';

const client = createClient({
  apiKey: process.env.API_KEY,
});

async function main() {
  // Create an item
  const item = await client.create({
    name: 'Hello World',
    description: 'My first item',
  });

  console.log('Created:', item.id);

  // Read it back
  const fetched = await client.get(item.id);
  console.log('Fetched:', fetched.name);

  // Update it
  const updated = await client.update(item.id, {
    name: 'Hello Universe',
  });
  console.log('Updated:', updated.name);

  // Delete it
  await client.delete(item.id);
  console.log('Deleted');
}

main().catch(console.error);
```

## With Express.js

Integrating with an Express server:

```javascript
import express from 'express';
import { createClient } from 'your-library';

const app = express();
const client = createClient({ apiKey: process.env.API_KEY });

app.use(express.json());

app.get('/items', async (req, res) => {
  try {
    const { items } = await client.list({
      limit: parseInt(req.query.limit) || 20,
    });
    res.json(items);
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

app.get('/items/:id', async (req, res) => {
  try {
    const item = await client.get(req.params.id);
    res.json(item);
  } catch (error) {
    res.status(404).json({ error: 'Not found' });
  }
});

app.post('/items', async (req, res) => {
  try {
    const item = await client.create(req.body);
    res.status(201).json(item);
  } catch (error) {
    res.status(400).json({ error: error.message });
  }
});

app.listen(3000, () => {
  console.log('Server running on port 3000');
});
```

## With React

Using the library in a React component:

```jsx
import { useState, useEffect } from 'react';
import { createClient } from 'your-library';

const client = createClient({ apiKey: process.env.REACT_APP_API_KEY });

function ItemList() {
  const [items, setItems] = useState([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);

  useEffect(() => {
    async function fetchItems() {
      try {
        const { items } = await client.list({ limit: 10 });
        setItems(items);
      } catch (err) {
        setError(err.message);
      } finally {
        setLoading(false);
      }
    }

    fetchItems();
  }, []);

  if (loading) return <div>Loading...</div>;
  if (error) return <div>Error: {error}</div>;

  return (
    <ul>
      {items.map(item => (
        <li key={item.id}>{item.name}</li>
      ))}
    </ul>
  );
}

export default ItemList;
```

## Error Handling

Comprehensive error handling example:

```javascript
import { createClient, LibraryError } from 'your-library';

const client = createClient({ apiKey: process.env.API_KEY });

async function safeOperation() {
  try {
    const item = await client.get('some-id');
    return item;
  } catch (error) {
    if (error instanceof LibraryError) {
      switch (error.code) {
        case 'NOT_FOUND':
          console.log('Item does not exist');
          return null;
        case 'UNAUTHORIZED':
          console.error('Check your API key');
          throw error;
        case 'RATE_LIMITED':
          console.log('Too many requests, waiting...');
          await new Promise(r => setTimeout(r, 1000));
          return safeOperation(); // Retry
        default:
          console.error(`Error: ${error.message}`);
          throw error;
      }
    }
    throw error;
  }
}
```

## TypeScript Example

Full TypeScript usage with custom types:

```typescript
import { createClient, type Config, type Item } from 'your-library';

interface CustomItem extends Item {
  metadata: {
    tags: string[];
    priority: 'low' | 'medium' | 'high';
  };
}

const config: Config = {
  apiKey: process.env.API_KEY!,
  debug: process.env.NODE_ENV === 'development',
};

const client = createClient(config);

async function createWithMetadata(): Promise<CustomItem> {
  return client.create<CustomItem>({
    name: 'Typed Item',
    metadata: {
      tags: ['example', 'typescript'],
      priority: 'high',
    },
  });
}
```

<Tip>
All examples assume you have set the `API_KEY` environment variable. Never hardcode API keys in your source code.
</Tip>
